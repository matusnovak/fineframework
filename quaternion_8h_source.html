180  inline void getEuler(T* Roll, T* Pitch, T* Yaw) const {
181  // http://www.euclideanspace.com/maths/geometry/rotations/conversions/QuatToEuler/
182  T test = x*y + z*w;
183  double heading;
184  double attitude;
185  double bank;
186  if (test > 0.499) { // singularity at north pole
187  heading = 2.0f * atan2(x, w);
188  attitude = M_PI_2;
189  bank = 0;
190  }
191  else if (test < -0.499) { // singularity at south pole
192  heading = -2.0f * atan2(x, w);
193  attitude = -M_PI_2;
194  bank = 0;
195  }
196  else {
197  float sqx = x * x;
198  float sqy = y * y;
199  float sqz = z * z;
200  heading = atan2(2.0f * y * w - 2.0f * x * z, 1.0f - 2.0f*sqy - 2.0f*sqz);
201  attitude = asin(2 * test);
202  bank = atan2(2.0f*x * w - 2.0f * y * z, 1.0f - 2.0f*sqx - 2.0f*sqz);
203  }
204  if (Roll != NULL)*Roll = T(attitude *DEG_TO_RAD);
205  if (Pitch != NULL)*Pitch = T(heading *DEG_TO_RAD);
206  if (Yaw != NULL)*Yaw = T(bank *DEG_TO_RAD);
207  }
