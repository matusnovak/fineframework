193  inline ffw::Vec3<T>& aotateByAxisRad(double rad, Vec3<T> axis) {
194  double sina = sin(rad);
195  double cosa = cos(rad);
196  double cosb = 1.0f - cosa;
197  auto v = *this;
198  x = static_cast<T>(v.x*(axis.x*axis.x*cosb + cosa) + v.y*(axis.x*axis.y*cosb - axis.z*sina) + v.z*(axis.x*axis.z*cosb + axis.y*sina));
199  y = static_cast<T>(v.x*(axis.y*axis.x*cosb + axis.z*sina) + v.y*(axis.y*axis.y*cosb + cosa) + v.z*(axis.y*axis.z*cosb - axis.x*sina));
200  z = static_cast<T>(v.x*(axis.z*axis.x*cosb - axis.y*sina) + v.y*(axis.z*axis.y*cosb + axis.x*sina) + v.z*(axis.z*axis.z*cosb + cosa));
201  return *this;
202  }

260  ffw::Vec3<T> qvc;

267  qvcc.z = static_cast<T>(quat.x*qvc.y - quat.y*qvc.x);
