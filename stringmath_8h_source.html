310  for (size_t i = 0; i < Wstr.size(); i++) {
311  const auto& w = Wstr[i];
312  // Only byte sequence
313  if (w <= 0x007F) {
314  if (Str != NULL && Str->size() > total) {

318  }

321  if (Str != NULL && Str->size() > total + 1) {
322  (*Str)[total + 0] = 0xC0 | char((w & 0x0FC0) >> 6);
323  (*Str)[total + 1] = 0x80 | char((w & 0x003F));
324  }
325  total += 2;
326  }
327  // Three byte sequence
328  else if (w <= 0xFFFF) {

331  (*Str)[total + 1] = 0x80 | char((w & 0x0FC0) >> 6);
332  (*Str)[total + 2] = 0x80 | char((w & 0x003F));
333  }
334  total += 3;
335  }
336  // Four byte sequence
337  else if (w <= 0x1FFFFF) {
338  if (Str != NULL && Str->size() > total + 3) {
339  (*Str)[total + 0] = 0xF0 | char((w & 0x1C0000) >> 18);
340  (*Str)[total + 1] = 0x80 | char((w & 0x03F000) >> 12);
341  (*Str)[total + 2] = 0x80 | char((w & 0x000FC0) >> 6);
342  (*Str)[total + 3] = 0x80 | char((w & 0x00003F));
343  }
344  total += 4;
345  }
346  // Five byte sequence
347  else if (w <= 0x3FFFFFF) {
348  if (Str != NULL && Str->size() > total + 3) {
349  (*Str)[total + 0] = 0xF8 | char((w & 0x3000000) >> 24);
350  (*Str)[total + 1] = 0x80 | char((w & 0x0FC0000) >> 18);
351  (*Str)[total + 2] = 0x80 | char((w & 0x003F000) >> 12);

376  size_t utf8ToWstr(WS* Wstr, const SS& Str) {
377  size_t total = 0;
378  wchar_t w = 0;
379  unsigned char bytes = 0;
380  unsigned char sequence = 0;
381  for (size_t i = 0; i < Str.size(); i++) {
382  const auto& c = Str[i];
383  // One byte in the sequence...
384  if (!(c & 0x80)) {
385  bytes = 0;
386  sequence = 0;
387  //w = c;
388  }
389  // Multiple bytes in the sequence...
390  else {
391  unsigned char b = 0x80;
392  bytes = 0;
393  while ((c & b) && bytes < 8) {
394  bytes++;
395  b = b >> 1;
396  }
397  if (bytes > 1)sequence = bytes;
398  }
399  // First byte 110x xxxx sequence of two bytes
400  if (bytes == 2 && sequence == 2) {
401  w = (c & 0x1F) << 6;
402  sequence--;
403  }
404  // First byte 1110 xxxx sequence of three bytes
405  else if (bytes == 3 && sequence == 3) {
406  w = (c & 0x0F) << 12;
407  sequence--;
408  }
409  // First byte 1111 0xxx sequence of four bytes
410  else if (bytes == 4 && sequence == 4) {
411  w = (c & 0x07) << 18;
412  sequence--;
413  }
414  // First byte 1111 10xx sequence of five bytes
415  else if (bytes == 5 && sequence == 5) {
416  w = (c & 0x03) << 24;
417  sequence--;
418  }
419  // First byte 1111 110x sequence of six bytes
420  else if (bytes == 6 && sequence == 6) {
421  w = (c & 0x01) << 30;
422  sequence--;
423  }
424  // Continuing byte 10xx xxxx
425  else if (bytes == 1 && sequence > 0) {
426  w |= (c & 0x3F) << ((sequence - 1) * 6);
427  sequence--;
428  // We have reached the end of the sequence
429  if (sequence == 0) {
430  if (Wstr != NULL && Wstr->size() > total) {
431  (*Wstr)[total] = w;
432  }
433  total++;
434  }
435  }
436  // No sequence
437  else {
438  w = c;
439  if (Wstr != NULL && Wstr->size() > total) {
440  (*Wstr)[total] = w;
441  }
442  total++;
443  //wstr.push_back(w);
444  }
445  }
446  return total;
447  }
