303  size_t wstrToUtf8(SS * Str, const WS& Wstr) {
304  size_t total = 0;
305  for (size_t i = 0; i < Wstr.size(); i++) {
306  const auto& w = Wstr[i];
307  // Only byte sequence
308  if (w <= 0x007F) {
309  if (Str != NULL && Str->size() > total) {
310  (*Str)[total + 0] = char(w & 0x007F);
311  }
312  total += 1;
313  }
314  // Two byte sequence
315  else if (w <= 0x7FF) {
316  if (Str != NULL && Str->size() > total + 1) {
317  (*Str)[total + 0] = 0xC0 | char((w & 0x0FC0) >> 6);
318  (*Str)[total + 1] = 0x80 | char((w & 0x003F));
319  }
320  total += 2;
321  }
322  // Three byte sequence
323  else if (w <= 0xFFFF) {
324  if (Str != NULL && Str->size() > total + 2) {
325  (*Str)[total + 0] = 0xE0 | char((w & 0xF000) >> 12);
326  (*Str)[total + 1] = 0x80 | char((w & 0x0FC0) >> 6);
327  (*Str)[total + 2] = 0x80 | char((w & 0x003F));
328  }
329  total += 3;
330  }
331  // Four byte sequence
332  else if (w <= 0x1FFFFF) {
333  if (Str != NULL && Str->size() > total + 3) {
334  (*Str)[total + 0] = 0xF0 | char((w & 0x1C0000) >> 18);
335  (*Str)[total + 1] = 0x80 | char((w & 0x03F000) >> 12);
336  (*Str)[total + 2] = 0x80 | char((w & 0x000FC0) >> 6);
337  (*Str)[total + 3] = 0x80 | char((w & 0x00003F));
338  }
339  total += 4;
340  }
341  // Five byte sequence
342  else if (w <= 0x3FFFFFF) {
343  if (Str != NULL && Str->size() > total + 3) {
344  (*Str)[total + 0] = 0xF8 | char((w & 0x3000000) >> 24);
345  (*Str)[total + 1] = 0x80 | char((w & 0x0FC0000) >> 18);
346  (*Str)[total + 2] = 0x80 | char((w & 0x003F000) >> 12);
347  (*Str)[total + 3] = 0x80 | char((w & 0x0000FC0) >> 6);
348  (*Str)[total + 4] = 0x80 | char((w & 0x000003F));
349  }
350  total += 5;
351  }
352  // Six byte sequence
353  else if (w <= 0x7FFFFFFF) {
354  if (Str != NULL && Str->size() > total + 3) {
355  (*Str)[total + 0] = 0xFC | char((w & 0x40000000) >> 30);
356  (*Str)[total + 1] = 0x80 | char((w & 0x3F000000) >> 24);
357  (*Str)[total + 2] = 0x80 | char((w & 0x00FC0000) >> 18);
358  (*Str)[total + 3] = 0x80 | char((w & 0x0003F000) >> 12);
359  (*Str)[total + 4] = 0x80 | char((w & 0x00000FC0) >> 6);
360  (*Str)[total + 5] = 0x80 | char((w & 0x0000003F));
361  }
362  total += 6;
363  }
364  }

371  size_t utf8ToWstr(WS* Wstr, const SS& Str) {
372  size_t total = 0;
373  wchar_t w = 0;
374  unsigned char bytes = 0;
375  unsigned char sequence = 0;
376  for (size_t i = 0; i < Str.size(); i++) {
377  const auto& c = Str[i];
378  // One byte in the sequence...
379  if (!(c & 0x80)) {
380  bytes = 0;
381  sequence = 0;
382  //w = c;
383  }
384  // Multiple bytes in the sequence...
385  else {
386  unsigned char b = 0x80;
387  bytes = 0;
388  while ((c & b) && bytes < 8) {
389  bytes++;
390  b = b >> 1;
391  }
392  if (bytes > 1)sequence = bytes;
393  }
394  // First byte 110x xxxx sequence of two bytes
395  if (bytes == 2 && sequence == 2) {
396  w = (c & 0x1F) << 6;
397  sequence--;
398  }
399  // First byte 1110 xxxx sequence of three bytes
400  else if (bytes == 3 && sequence == 3) {
401  w = (c & 0x0F) << 12;
402  sequence--;
403  }
404  // First byte 1111 0xxx sequence of four bytes
405  else if (bytes == 4 && sequence == 4) {
406  w = (c & 0x07) << 18;
407  sequence--;
408  }
409  // First byte 1111 10xx sequence of five bytes
410  else if (bytes == 5 && sequence == 5) {
411  w = (c & 0x03) << 24;
412  sequence--;
413  }
414  // First byte 1111 110x sequence of six bytes
415  else if (bytes == 6 && sequence == 6) {
416  w = (c & 0x01) << 30;
417  sequence--;
418  }
419  // Continuing byte 10xx xxxx
420  else if (bytes == 1 && sequence > 0) {
421  w |= (c & 0x3F) << ((sequence - 1) * 6);
422  sequence--;
423  // We have reached the end of the sequence
424  if (sequence == 0) {
425  if (Wstr != NULL && Wstr->size() > total) {
426  (*Wstr)[total] = w;
427  }
428  total++;
429  }
430  }
431  // No sequence
432  else {
433  w = c;
434  if (Wstr != NULL && Wstr->size() > total) {
435  (*Wstr)[total] = w;
436  }
437  total++;
438  //wstr.push_back(w);
439  }
440  }
441  return total;
442  }
